To use rounded tabs:

    .tab-position .tab:not(.tab-group) {
    	border-radius: 5px 5px 0px 0px;
    	box-shadow: 0px -1px #ccc;
    }
    
    .stacks-on .tab-strip .tab .tab-group-indicator .tab-indicator {
    	border-radius: 4px 4px 0 0/50% 50% 0 0;
    	border-top: 1px solid #bbb;
    }
